const int N = 1e5 + 10;
struct Range {
    int l, r;
    bool operator < (const Range &t) const {
        return l < t.l;
    }
} range[N];
int n, S, T;

int main() {
    scanf("%d%d%d", &S, &T, &n);
    for (int i = 0; i < n; ++ i)   scanf("%d%d", &range[i].l, &range[i].r);
    sort(range, range + n);
    
    bool ok = false;
    int ans = 0;
    for (int i = 0; i < n; ++ i) {
        int j = i, r = -2e9;
        while (j < n && range[j].l <= S) {
            r = max(r, range[j].r);
            ++ j;
        }
        
        if (r < S) {
            ans = -1;
            break;
        }
        
        ++ ans;
        
        if (r >= T) {
            ok = true;
            break;
        }
        
        S = r, i = j - 1;
    }
    
    if (ok) printf("%d\n", ans);
    else    puts("-1");
    
    return 0;
}

    scanf("%d", &n);

    }

}
